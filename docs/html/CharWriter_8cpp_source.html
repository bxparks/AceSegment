180 void CharWriter::writeCharAt(uint8_t digit, char c,
181  StyledDigit::StyleType style) {
182  if (digit >= getNumDigits()) return;
183  uint8_t pattern = ((uint8_t) c < kNumCharacters)
184  ? pgm_read_byte(&kCharacterArray[(uint8_t) c])
185  : UNKNOWN;
186  mRenderer->writePatternAt(digit, pattern, style);
187 }
