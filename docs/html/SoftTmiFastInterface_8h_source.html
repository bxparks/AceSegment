67 template <
68  uint8_t T_CLK_PIN,
69  uint8_t T_DIO_PIN,
70  uint16_t T_DELAY_MICROS
71 >
72 class SoftTmiFastInterface {
73  public:
74  explicit SoftTmiFastInterface() = default;
75 
77  void begin() const {
78  // These are open-drain lines, with a pull-up resistor. We must not drive
79  // them HIGH actively since that could damage the transitor at the other
80  // end of the line pulling LOW. Instead, we go into INPUT mode to let the
81  // line to HIGH through the pullup resistor, then go to OUTPUT mode only
82  // to pull down.
83  digitalWriteFast(T_CLK_PIN, LOW);
84  digitalWriteFast(T_DIO_PIN, LOW);
85 
86  // Begin with both lines at HIGH.
87  clockHigh();
88  dataHigh();
89  }
90 
92  void end() const {
93  clockHigh();
94  dataHigh();
95  }
96 
98  void startCondition() const {
99  clockHigh();
100  dataHigh();
101 
102  dataLow();
103  clockLow();
104  }
105 
107  void stopCondition() const {
108  dataLow();
109  clockHigh();
110  dataHigh();
111  }
112 
117  uint8_t sendByte(uint8_t data) const {
118  for (uint8_t i = 0; i < 8; ++i) {
119  if (data & 0x1) {
120  dataHigh();
121  } else {
122  dataLow();
123  }
124  clockHigh();
125  clockLow();
126  data >>= 1;
127  }
128 
129  // Device places the ACK/NACK bit upon the falling edge of the 8th CLK,
130  // which happens in the loop above.
131  pinModeFast(T_DIO_PIN, INPUT);
132  bitDelay();
133  uint8_t ack = digitalReadFast(T_DIO_PIN);
134 
135  // Device releases DIO upon falling edge of the 9th CLK.
136  clockHigh();
137  clockLow();
138  return ack;
139  }
140 
141  private:
142  void bitDelay() const { delayMicroseconds(T_DELAY_MICROS); }
143 
144  void clockHigh() const { pinModeFast(T_CLK_PIN, INPUT); bitDelay(); }
145 
146  void clockLow() const { pinModeFast(T_CLK_PIN, OUTPUT); bitDelay(); }
147 
148  void dataHigh() const { pinModeFast(T_DIO_PIN, INPUT); bitDelay(); }
149 
150  void dataLow() const { pinModeFast(T_DIO_PIN, OUTPUT); bitDelay(); }
151 };
