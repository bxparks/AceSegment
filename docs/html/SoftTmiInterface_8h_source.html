77  void begin() const {
78  // These are open-drain lines, with a pull-up resistor. We must not drive
79  // them HIGH actively since that could damage the transitor at the other
80  // end of the line pulling LOW. Instead, we go into INPUT mode to let the
81  // line to HIGH through the pullup resistor, then go to OUTPUT mode only
82  // to pull down.
83  digitalWrite(mClkPin, LOW);
84  digitalWrite(mDioPin, LOW);
85 
86  // Begin with both lines at HIGH.
87  clockHigh();
88  dataHigh();
89  }
90 
92  void end() const {
93  clockHigh();
94  dataHigh();
95  }

104  }

107  void stopCondition() const {

109  clockHigh();

117  uint8_t sendByte(uint8_t data) const {
118  for (uint8_t i = 0; i < 8; ++i) {
119  if (data & 0x1) {
120  dataHigh();
121  } else {
122  dataLow();
123  }
124  clockHigh();
125  clockLow();
126  data >>= 1;
127  }
128 
129  // Device places the ACK/NACK bit upon the falling edge of the 8th CLK,
130  // which happens in the loop above.
131  pinMode(mDioPin, INPUT);
132  bitDelay();
133  uint8_t ack = digitalRead(mDioPin);
134 
135  // Device releases DIO upon falling edge of the 9th CLK.
136  clockHigh();
137  clockLow();
138  return ack;
139  }
140 
141  private:
142  void bitDelay() const { delayMicroseconds(mDelayMicros); }
143 
144  void clockHigh() const { pinMode(mClkPin, INPUT); bitDelay(); }
145 
146  void clockLow() const { pinMode(mClkPin, OUTPUT); bitDelay(); }
147 
148  void dataHigh() const { pinMode(mDioPin, INPUT); bitDelay(); }
149 
150  void dataLow() const { pinMode(mDioPin, OUTPUT); bitDelay(); }
151 
152  private:
153  uint8_t const mClkPin;
154  uint8_t const mDioPin;
155  uint16_t const mDelayMicros;
156 };
