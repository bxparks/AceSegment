102  bool scrollRight() {
103  uint8_t numDigits = display().getNumDigits();
104  bool isDone = (mStringPos <= -numDigits);
105  if (! isDone) {

112  private:
113  // disable copy-constructor and assignment operator
114  StringScroller(const StringScroller&) = delete;
115  StringScroller& operator=(const StringScroller&) = delete;
116 
117  void writeString() {
118  uint8_t numDigits = display().getNumDigits();
119  int16_t stringPos = mStringPos;
120  for (uint8_t pos = 0; pos < numDigits; pos++) {
121  char c;
122  if (stringPos < 0 || stringPos >= mStringLength) {
123  c = ' ';
124  } else {
125  // Normally inserting an if-conditional inside a for-loop is not a
126  // good idea. But this is not a speed-critical loop and it loops over
127  // only the number of digits, which will almost always be <= 8. So I
128  // think this is better than creating a duplicate version of this
129  // function using a template function.
130  if (mIsFlashString) {
131  c = pgm_read_byte((const uint8_t*) mString + stringPos);
132  } else {
133  c = *((const char*) mString + stringPos);
134  }
135  }
136  mCharWriter.writeCharAt(pos, c);
137  stringPos++;
138  }
139  }
140 
141  private:
142  // The order of these fields is partially motivated to reduce memory
143  // consumption on 32-bit processors.
144  CharWriter& mCharWriter;
145  const void* mString;
146  int16_t mStringPos; // can become negative
147  uint8_t mStringLength;
148  bool mIsFlashString;
